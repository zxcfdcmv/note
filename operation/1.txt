{ config, pkgs, lib, userSettings, ... }:

let
  finalKwmConfig = pkgs.substituteAll {
    src = ../../assets/kwm-config.zon;
    
    # 定义占位符对应的真实路径
    # 注意：变量名必须与 .zon 里的 @变量名@ 一致
    stylix_image = "${config.stylix.image}";
    swaybg = "${pkgs.swaybg}/bin/swaybg";
    footserver = "${pkgs.foot}/bin/foot --server";
    footclient = "${pkgs.foot}/bin/footclient";
    tofi = "${pkgs.tofi}/bin/tofi-run";
    
    fg = config.lib.stylix.colors.base05;
    bg = config.lib.stylix.colors.base00;
    accent = config.lib.stylix.colors.base0D;
    dim = config.lib.stylix.colors.base03;
    font = "Maple Mono NF CN";
  };

  kwm = pkgs.zig.buildZigPackage rec {
    pname = "kwm";
    version = "master";
    src = pkgs.fetchzip {
      url = "https://gh-proxy.com/https://github.com/kewuaa/kwm/archive/refs/heads/master.zip";
      sha256 = lib.fakeHash;
      stripRoot = true;
    };

    zigDepsHash = lib.fakeHash;

    nativeBuildInputs = [
      pkgs.pkg-config
      pkgs.wayland-scanner
    ];

    buildInputs = [
      pkgs.wayland
      pkgs.wayland-protocols
      pkgs.libxkbcommon
      pkgs.pixman
      pkgs.fcft
    ];

    postPatch = ''
      cp ${finalKwmConfig} config.zon
    '';

    optimize = "ReleaseSafe";

    zigBuildFlags = [ "-Dconfig=config.zon" ];
  };

in {
  # --- 系统级配置 ---
  environment.systemPackages = [
    pkgs.river
    kwm
  ];

  services.greetd = {
    enable = true;
    settings.default_session.command = ''
      ${pkgs.tuigreet}/bin/tuigreet \
      --cmd "${pkgs.river}/bin/river -c ${kwm}/bin/kwm" \
      --theme "dark" --remember
    '';
  };

  xdg.portal = {
    enable = true;
    extraPortals = with pkgs; [
      xdg-desktop-portal-wlr
      xdg-desktop-portal-gtk
    ];
    config = {
      river = {
        default = [ "wlr" "gtk" ];
      };
      common = {
        default = [ "wlr" ];
      };
    };
  };
  # --- 用户级配置 (通过 HM 注入) ---
  home-manager.users.${userSettings.username} = { ... }: {
    home.packages = [
      pkgs.tofi
      pkgs.foot
      pkgs.wl-clipboard-rs
    ];
  };
}
